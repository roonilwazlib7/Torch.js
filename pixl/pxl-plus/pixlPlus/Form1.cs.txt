        private void button1_Click(object sender, EventArgs e)
        {
            Bitmap im = new Bitmap(textBox1.Text);
            int width = im.Width;
            int height = im.Height;
            int uniqueColors = 0;
            string output = "var " + outName.Text + " = [\n";
            string output2 = "var " + outName.Text + "Pallette = {\n";
            Dictionary<string, string> colorMap = new Dictionary<string, string>();
            Dictionary<string, string> colorPallette = new Dictionary<string, string>();
            for (int i = 0; i < width; i++)
            {
                output += "\t'";
                for (int j = 0; j < height; j++)
                {
                    Color pixelColor = im.GetPixel(i, j);
                    if (pixelColor.A == 0)
                    {
                        output += ".";
                        if (HigherBits.Checked)
                        {
                            output += "~";
                        }
                    }
                    else
                    {
                        string rgba = pixelColor.R + "," + pixelColor.G + "," + pixelColor.B + "," + pixelColor.A;
                        if (colorMap.ContainsKey(rgba))
                        {
                            if (HigherBits.Checked)
                            {
                                output += colorMap[rgba] + "~";
                            }
                            else
                            {
                                output += colorMap[rgba];
                            }                         
                        }
                        else
                        {
                            uniqueColors++;
                            if (HigherBits.Checked)
                            {
                                output += getPixel(uniqueColors) + "~";
                            }
                            else
                            {
                                output += getPixel(uniqueColors);
                            }
                            colorMap[rgba] = uniqueColors.ToString();
                            colorPallette[getPixel(uniqueColors)] = "rgba(" + rgba.ToString() + ")";
                        }
                    }
                    
                }
                output += "',";
                output += "\n";
            }
            output += "];";
            outputBox.Text = output;

            foreach(KeyValuePair<string,string> pair in colorPallette)
            {
                output2 += "\t'" + pair.Key + "' : '" + pair.Value + "',\n";
            }
            output2 += "};";
            outputBox2.Text = output2;
        }
        private string getPixel(int num)
        {
            string index = "abcdefghijklmnopqrstuvwxyz";
            index += index.ToUpper();
            if (num > 9 && num <= 61)
            {
                int pos = num - 10;
                return index[pos].ToString();
            }
            else
            {
                return num.ToString();
            }
        }
